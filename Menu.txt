    public final double PLAIN_BAGEL = 1.25;
    public final double WHEAT_BAGEL = 1.50;
    public final double EVERYTHING_BAGEL = 1.75;
    public final double CREAM_CHEESE = .50;
    public final double BUTTER = .25;
    public final double PEACH_JELLY = .75;
    public final double BLUEBERRY_JAM = .75;

    public double getBagelCost(){
        double bagelCost = 0.0;
        if(plainBagel.isSelected()){
            bagelCost = PLAIN_BAGEL;
        } else if(wheatBagel.isSelected()){
            bagelCost = WHEAT_BAGEL;
        } else if(everythingBagel.isSelected()){
            bagelCost = EVERYTHING_BAGEL;
        }
        return bagelCost;
    }//close getBagelCost method
    
    public double getToppingCost(){
        double toppingCost = 0.0;
        if(creamCheese.isSelected()){
            toppingCost += CREAM_CHEESE;
        } 
        if(butter.isSelected()){
            toppingCost += BUTTER;
        }
        if(peachJelly.isSelected()){
            toppingCost += PEACH_JELLY;
        }
        if(blueberryJam.isSelected()){
            toppingCost += BLUEBERRY_JAM;
        }
        return toppingCost;
    }//close getBagelCost method
